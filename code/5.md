# 6. C语言之运算符练习题

```c
#include <stdio.h>
int main()
{
    float r, s, c;
    /*********begin*********/
    scanf("%f", &r);
    s = 3.14 * r * r;
    c = 2 * 3.14 * r;
    printf("请输入圆半径r:圆面积S=%.6f,圆周长C=%.6f", s, c);
    /*********end*********/
    return 0;
}
```

```c
#include <stdio.h>
int main()
{
    double t, a, v0, v, s;
    /*********begin*********/
    scanf("%lf %lf %lf", &t, &a, &v0);
    v = v0 + a * t;
    s = v0 * t + 0.5 * a * t * t;
    printf("当初始速度为%.6lf时\n", v0);
    printf("%.6lfs时的速度为%.6lf\n", t, v);
    printf("%.6lfs时的位移为%.6lf", t, s);
    /*********end*********/
    return 0;
}
```

```c
#include <stdio.h>
#include <math.h>
int main()
{
    /*********begin*********/
    float x1, y1, x2, y2;
    scanf("%f %f %f %f", &x1, &y1, &x2, &y2);
    float res = sqrt((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
    printf("%.2f", res);
    /*********end*********/
    return 0;
}
```
