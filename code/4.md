# 5. C语言——运算符和表达式

```c
#include <stdio.h>
int main()
{
    float t, a, v0, v, s;
    scanf("%f%f%f", &t, &a, &v0);
    /********** Begin *********/
    v = v0 + a * t;
    s = v0 * t + 0.5 * a * t * t;
    printf("当初始速度为%.2f时\n", v0);
    printf("%.2fs时的速度为%.2f\n", t, v);
    printf("%.2fs时的位移为%.2f\n", t, s);
    /********** End **********/
    return 0;
}
```

```c
#include <stdio.h>
int main()
{
    int a, b, c, d, m, n;
    scanf("%d", &a);
    /********** Begin *********/
    b = a / 1000 % 10;
    c = a / 100 % 10;
    d = a / 10 % 10;
    m = a % 10;
    n = m * 1000 + d * 100 + c * 10 + b;
    /********** End **********/
    printf("重新组合后:%d", n);
    return 0;
}
```
