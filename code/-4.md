void func(int n,int count){
    if(n/10){
        func(n/10,count+1);
        if(count==0)printf("%d\n",n%10);
        else printf("%d ",n%10);
    }else{
        printf("%d\n%d ",count+1,n%10);
    }
}

int put(int n,int z){
    if(n){
        if(z!=0||n%10!=0)printf("%d",n%10),z=1;
        put(n/10,z);
    }
}

int main(){
    int n;
    scanf("%d",&n);
    func(n,0);
    put(n,0);

```c
#include<stdio.h>

int is_prime(int n){
    if(n<2)return 0;
    for(int i=2;i<n;i++){
        if(n%i==0)return 0;
    }
    return 1;
}

int main(){
    int n,k,sum=0,cnt=0;
    scanf("%d %d",&n,&k);
    for(int i=n;i>1;i--){
        if(is_prime(i)){
            if(cnt!=0)printf("+");
            printf("%d",i);
            sum+=i;
            cnt++;
            if(cnt==k)break;
        }
    }
    printf("=%d",sum);
}
```
