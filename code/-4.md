```c
#include<stdio.h>

int main(){
    int n,ans=0,sum=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        ans*=10;
        ans+=2;
        sum+=ans;
    }
    printf("%d",sum);
}
```

```c
#include<stdio.h>

int main(){
    int a,b,c;
    double sum=0;
    scanf("%d %d %d",&a,&b,&c);
    for(int i=1;i<=a;i++)sum+=i;
    for(int i=1;i<=b;i++)sum+=i*i;
    for(int i=1;i<=c;i++)sum+=1.0/i;
    printf("%.2f",sum);
}
```

```c
#include<stdio.h>

int main(){
```
```c
    int n;
    int a[20000];
    a[0]=0;a[1]=0;
    for(int i=2;i<20000;i++)a[i]=1;
    for(int i=2;i<20000;i++){
        if(a[i]){
            for(int j=i*2;j<20000;j+=i)a[j]=0;
        }
    }
    scanf("%d",&n);
    int cnt=0;
    for(int i=n-1;i>0&&cnt<10;i--){
        if(a[i]==1){
            printf("%6d",i);
            cnt++;
        }
    }
}
```
