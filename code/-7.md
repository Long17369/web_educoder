# 7.C语言集训_递归函数（2024）

```c
#include<stdio.h>
typedef long long ll;

ll func(ll n){
    if(n==0LL)return 1LL;
    return n*func(n-1);
}

int main(){
    ll n;
    scanf("%lld",&n);
    printf("%lld",func(n));
}
```

```c
#include<stdio.h>
typedef long long ll;

ll sum(ll a,ll b){
    ll sum=0;
    for(ll i=a;i<=b;i++)sum+=i;
    return sum;
}

int main(){
    ll a,b;
    scanf("%lld %lld",&a,&b);
    printf("%lld",sum(a,b));
}
```

```c
#include<stdio.h>
typedef long long ll;

ll fibo(ll N){
    if(N==0LL)return 0LL;
    else if(N==1LL)return 1LL;
    return fibo(N-1)+fibo(N-2);
}

int main(){
    ll N;
    scanf("%lld",&N);
    printf("%lld",fibo(N));
}
```

```c
#include<stdio.h>
typedef long long ll;

ll gcd(ll a,ll b){
    if(b==0LL)return a;
    return gcd(b,a%b);
}

int main(){
    ll a,b;
    scanf("%lld%lld",&a,&b);
    printf("%lld",gcd(a,b));
}
```

```c
#include<stdio.h>
typedef long long ll;

void o(ll n){
    if(n==0)return;
    o(n/2);
    printf("%lld",n&1LL);
}

int main(){
    ll a,b;
    scanf("%lld %lld",&a,&b);
    for(ll i=a;i<=b;i++){
        o(i);
        printf("\n");
    }
}
```
