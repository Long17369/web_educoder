# 26. 字符数组及其字符串应用

```c
#include "stdio.h"
#include "string.h"

int main()
{
    char s[80];
    int i, j;
    scanf("%s", s);
    printf("The original string: \n");
    puts(s);
    /***** 以下一行有错误 *****/
    for (i = j = 0; s[i] != '\0'; i++)
    {
        if (s[i] != 'c')
            /***** 以下一行有错误 *****/
            s[j++] = s[i];
    }
    /***** 以下一行有错误 *****/
    s[j] = '\0';
    printf("The string after deleted: \n");
    puts(s);
    return 0;
}
```

```c
    int cnt=0;
    char ch;
    while((ch=getchar())!=EOF&&ch!='\n'){
        cnt++;
        if(cnt%2==0||ch%2==0){
            putchar(ch);
        }
    }
}
```

```c
#include <stdio.h>
#include <string.h>
int main() {
    int len=0,n;
    char *max;
    scanf("%d", &n);
    char str[n][100];
    while (n--)
    {
        scanf("%s[^\n]", str[n]);
        if(len<strlen(str[n])){
            len=strlen(str[n]);
            max=str[n];
        }
    }
    puts(max);
}
```

```c
/********** Begin **********/
#include <stdio.h>
#include <string.h>

int main() {
    char str[9][10];
    char *p[9];
    for(int i=0;i<9;i++)p[i]=str[i];
    for(int i=0;i<9;i++)scanf("%s",str[i]);
    for(int i=0;i<8;i++){
        for(int j=0;j<8;j++){
            if(strcmp(p[j],p[j+1])>0){
                char *temp=p[j];
                p[j]=p[j+1];
                p[j+1]=temp;
            }
        }
    }
    puts("排序前的字符串：");
    for(int i=0;i<9;i++)puts(str[i]);
    puts("排序后的字符串：");
    for(int i=0;i<9;i++)puts(p[i]);
    return 0;
}
/********** End **********/
```
