#include<stdio.h>

int prime(int n){
    if(n<2)return 0;
    for(int i=2;i<n;i++)
        if(n%i==0)return 0;
    return 1;
}

int func(int n){
    if(n==0)return 1;
    int a[1001],cnt=0;
    while(n!=0){
        a[cnt++]=n%10;
        n/=10;
    }
    if(cnt>5)return 0;
    int l=0,r=cnt-1;
    while(l<r){
        if(a[l++]!=a[r--])return 0;
    }
    return 1;
}

int main(){
    int num;
    while(scanf("%d",&num)==1){
        if(func(num)&&prime(num))puts("Yes");
        else puts("No");
    }
}

        if(n%i==0)return 0;
    return 1;

int sum_prime(int n){
    int sum=0;
    while(n){
        sum+=n%10;
        n/=10;
    }
    return prime(sum);
}

int main(){
    int num;
    while(scanf("%d",&num),num!=0){
        if(prime(num)&&sum_prime(num))printf("%d\n",num);
    }
