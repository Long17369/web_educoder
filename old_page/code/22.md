# 23. 练习-插入排序

```c
#include <stdio.h>
#include<stdlib.h>
#define N 100
int main ()
{ 
    /********** Begin **********/ 
    int n;
    scanf("%d", &n);
    int a[n+1];
    for (int i = 0; i < n; i++) scanf("%d", &a[i]);
    int x;
    scanf("%d", &x);
    for(int i=n-1;i>=0;i--){
        if(a[i]>x){
            a[i+1]=a[i];
        }
        else{
            a[i+1]=x;
            break;
        }
        if(i==0){
            a[0]=x;
        }
    }
    for (int i = 0; i <= n; i++) printf("%d ", a[i]);
    /********** End **********/
   return 0;
}
```

```c
#include <stdio.h>
#include<stdlib.h>
#define N 100
int main ()
{
    /********** Begin **********/
    int a[N],n, i, j, key;
    scanf("%d", &n);
    for (i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    for (i = 1; i < n; i++) {
        key = a[i];
        j = i - 1;
        while (j >= 0 && a[j] > key) {
            a[j + 1] = a[j];
            j--;
        }
        a[j + 1] = key;
        for (int k = 0; k < n; k++) printf("%d ", a[k]);
        printf("\n");
    }
    /********** End **********/
    return 0;
}
```
