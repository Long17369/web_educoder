```java
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int row = sc.nextInt();
        int min_num = Integer.MAX_VALUE;
        int[][] nums = new int[row][];
        for (int i = 0; i < row; i++) {
            int col = sc.nextInt();
            nums[i] = new int[col];
            for (int j = 0; j < col; j++) {
                nums[i][j] = sc.nextInt();
                min_num = Math.min(min_num, nums[i][j]);
            }
        }
        for (int[] is : nums) {
            for (int i = 0; i < is.length; i++) {
                is[i] -= min_num;
                System.out.print(is[i]);
                if (i != is.length - 1) {
                    System.out.print(" ");
                } else {
                    System.out.println();
                }
            }
        }
        sc.close();
    }
}
```
