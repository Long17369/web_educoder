```python3
N = int(input())

for i in range(N):
    try:
        x = input().split(".",3)
        assert all(str(int(i)) == i for i in x)
        x1,x2,x3,x4 = map(int,x)
        if x1 == 0:
            assert x2 == 0 and x3 == 0 and x4 == 0
        else:
            assert x1 in range(256) and x2 in range(256) and x3 in range(256) and x4 in range(256)
    except Exception:
        print("NO")
    else:
        print("YES")
```
