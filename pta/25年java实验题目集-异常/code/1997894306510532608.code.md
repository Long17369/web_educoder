import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        try {
            while (sc.hasNext()) {
                String cmd = sc.next();
                if (!"new".equals(cmd)) {
                    break;
                }
                String line = sc.nextLine();
                if (line.trim().isEmpty()) {
                    if (sc.hasNextLine()) {
                        line = sc.nextLine();
                    } else {
                        break;
                    }
                }
                Scanner lineSc = new Scanner(line);
                try {
                    String name = lineSc.next();
                    int scoreToAdd = lineSc.nextInt();
                    Student s = new Student();
                    s.setName(name);
                    s.addScore(scoreToAdd);
                    System.out.println(s);
                } catch (IllegalNameException e) {
                    System.out.println(e);
                } catch (IllegalScoreException e) {
                    System.out.println(e);
                } catch (Exception e) {
                    System.out.println(e);
                } finally {
                    lineSc.close();
                }
            }
        } finally {
            sc.close();
            System.out.println("scanner closed");
        }
    }
}

class IllegalNameException extends RuntimeException {
    public IllegalNameException(String message) {
        super(message);
    }
}

class IllegalScoreException extends Exception {
    public IllegalScoreException(String message) {
        super(message);
    }
}

class Student {
    private String name;
    private int score;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        if (name != null && !name.isEmpty() && Character.isDigit(name.charAt(0))) {
            throw new IllegalNameException("the first char of name must not be digit, name=" + name);
        }
        this.name = name;
    }

    public int getScore() {
        return score;
    }

    public void setScore(int score) {
        this.score = score;
    }

    public int addScore(int score) throws IllegalScoreException {
        int newScore = this.score + score;
        if (newScore < 0 || newScore > 100) {
            throw new IllegalScoreException("score out of range, score=" + newScore);
        }
        this.score = newScore;
        return this.score;
    }

    @Override
    public String toString() {
        return "Student [name=" + name + ", score=" + score + "]";
    }
}
