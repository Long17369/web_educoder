import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int m = 0, n = 0;
        int r = 0;
        try {
            m = sc.nextInt();
            n = sc.nextInt();
            r = sc.nextInt();
        } catch (Exception e) {
            System.out.println("data type error");
            sc.close();
            return;
        }
        if (r != 90 && r != 180 && r != -90) {
            System.out.println("angle data error");
            sc.close();
            return;
        }
        int[][] matrix = new int[m][n];
        try {
            for (int i = 0; i < m; i++) {
                for (int j = 0; j < n; j++) {
                    matrix[i][j] = sc.nextInt();
                    if (matrix[i][j] < 0 || matrix[i][j] > 255) {
                        System.out.println("matrix data error");
                        sc.close();
                        return;
                    }
                }
            }
        } catch (Exception e) {
            System.out.println("data type error");
            return;
        }
        sc.close();
        int[][] resultMatrix = null;
        if (r == 90) {
            resultMatrix = rotate90(matrix, m, n);
        } else if (r == -90) {
            resultMatrix = rotateMinus90(matrix, m, n);
        } else if (r == 180) {
            resultMatrix = rotate180(matrix, m, n);
        }
        for (int i = 0; i < resultMatrix.length; i++) {
            for (int j = 0; j < resultMatrix[i].length; j++) {
                System.out.print(resultMatrix[i][j]);
                if (j != resultMatrix[i].length - 1) {
                    System.out.print(" ");
                }
            }
            System.out.println();
        }
    }

    public static int[][] rotateMinus90(int[][] matrix, int m, int n) {
        int[][] result = new int[n][m];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                result[j][m - 1 - i] = matrix[i][j];
            }
        }
        return result;
    }

    public static int[][] rotate90(int[][] matrix, int m, int n) {
        int[][] result = new int[n][m];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                result[n - 1 - j][i] = matrix[i][j];
            }
        }
        return result;
    }

    public static int[][] rotate180(int[][] matrix, int m, int n) {
        int[][] result = new int[m][n];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                result[m - 1 - i][n - 1 - j] = matrix[i][j];
            }
        }
        return result;
    }
}
