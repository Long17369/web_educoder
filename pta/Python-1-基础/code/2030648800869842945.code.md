```python3
digits = ["零", "一", "二", "三", "四", "五", "六", "七", "八", "九"]
units = ["千", "百", "十", ""]
group_units = ["", "万", "亿"]


def read_group(val: int, is_highest: bool) -> str:
    if val == 0:
        return ""
    d = [0, 0, 0, 0]
    tmp = val
    for i in range(3, -1, -1):
        d[i] = tmp % 10
        tmp //= 10
    res = []
    prev_zero = False
    for idx in range(4):
        cur = d[idx]
        if cur == 0:
            prev_zero = True
            continue
        if prev_zero and res:
            res.append("零")
        if idx == 2 and cur == 1 and is_highest and d[0] == 0 and d[1] == 0:
            res.append("十")
        else:
            res.append(digits[cur] + units[idx])
        prev_zero = False
    return "".join(res)


def convert(s: str) -> str:
    s = s.lstrip("0") or "0"
    if s == "0":
        return "零"
    parts = []
    while s:
        parts.append(int(s[-4:]))
        s = s[:-4]
    highest = -1
    for i in range(len(parts) - 1, -1, -1):
        if parts[i] != 0:
            highest = i
            break
    if highest == -1:
        return "零"
    out = []
    zero_between = False
    for i in range(highest, -1, -1):
        g = parts[i]
        if g == 0:
            zero_between = True
            continue
        if zero_between or g < 1000:
            out.append("零")
            zero_between = False
        grp = read_group(g, is_highest=(i == highest))
        out.append(grp + (group_units[i] if i < len(group_units) else ""))
    return "".join(out).lstrip("零")


n = int(input())
for _ in range(n):
    s = input()
    print(convert(s))
```
