```python3
a = input().strip()
b = input().strip()
n = len(a)
m = len(b)

pre = [0] * (m + 1)
for i in range(m):
    pre[i + 1] = pre[i] + (b[i] == '1')

total = 0
for j in range(n):
    left = j
    right = j + (m - n)
    ones = pre[right + 1] - pre[left]
    zeros = (right - left + 1) - ones
    if a[j] == '0':
        total += ones
    else:
        total += zeros

print(total)
```
