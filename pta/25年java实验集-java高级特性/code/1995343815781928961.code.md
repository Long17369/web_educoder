import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        MyDate d = new MyDate(sc.nextInt(), sc.nextInt(), sc.nextInt());
        d.addDays(sc.nextInt());
        sc.close();
        System.out.println(d);
    }
}

class MyDate {
    private int year;
    private int month;
    private int day;
    private static final int[] DAYS = { 0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

    public MyDate(int year, int month, int day) {
        this.year = year;
        this.month = month;
        this.day = day;
    }

    public void addDays(int days) {
        this.day += days;
        while (this.day > getDaysInMonth(this.year, this.month)) {
            this.day -= getDaysInMonth(this.year, this.month);
            this.month++;
            if (this.month > 12) {
                this.month = 1;
                this.year++;
            }
        }
    }

    private int getDaysInMonth(int year, int month) {
        if (month == 2 && isLeapYear(year)) {
            return 29;
        }
        return DAYS[month];
    }

    private boolean isLeapYear(int year) {
        return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
    }

    @Override
    public String toString() {
        return String.format("%04d年%02d月%02d日", year, month, day);
    }
}
