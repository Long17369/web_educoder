import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        List<MobilePhone> phones = new ArrayList<>();
        for (int i = 0; i < 3; i++) {
            String type = sc.next();
            int price = sc.nextInt();
            phones.add(new MobilePhone(type, price));
        }
        System.out.println("排序前，链表中的数据：");
        for (MobilePhone phone : phones) {
            System.out.println(phone.getInfo());
        }
        phones.sort((p1, p2) -> p1.compareTo(p2));
        System.out.println("排序后，链表中的数据：");
        for (MobilePhone phone : phones) {
            System.out.println(phone.getInfo());
        }
        String type = sc.next();
        int price = sc.nextInt();
        sc.close();
        MobilePhone newPhone = new MobilePhone(type, price);
        for (MobilePhone phone : phones) {
            if (phone.compareTo(newPhone) == 0) {
                System.out.println(newPhone.type + "与链表中的" + phone.type + "价格相同");
                return;
            }
        }
        System.out.println("链表中的对象，没有一个与" + newPhone.type + "价格相同的");
    }
}

class MobilePhone implements Cloneable {
    public String type;
    public int price;

    public MobilePhone(String type, int price) {
        this.type = type;
        this.price = price;
    }

    public int compareTo(MobilePhone mp) {
        return this.price - mp.price;
    }

    public String getInfo() {
        return "型号：" + type + "，价格：" + price;
    }
}
