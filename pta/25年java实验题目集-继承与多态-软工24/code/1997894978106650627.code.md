import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String bankType = scanner.next();
        double amount = scanner.nextDouble();
        Customer customer = new Customer();
        IBank bank;
        if (bankType.equals("ABC")) {
            bank = new ABC();
        } else if (bankType.equals("ICBC")) {
            bank = new ICBC();
        } else {
            scanner.close();
            return;
        }
        double money = customer.getMoneyFromBank(bank, amount);
        System.out.printf("%.1f", money);
        scanner.close();
    }
}

class Customer {
    Customer() {
    }

    public double getMoneyFromBank(IBank bank, double money) {
        return bank.withdraw(money);
    }
}

class ABC implements IBank {
    @Override
    public double withdraw(double money) {
        return money * 1.002;
    }
}

class ICBC implements IBank {
    @Override
    public double withdraw(double money) {
        return money * 1.001;
    }
}

interface IBank {
    double withdraw(double money);
}
