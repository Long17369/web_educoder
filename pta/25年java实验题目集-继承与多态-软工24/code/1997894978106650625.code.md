```java
public class Main {
    public static void main(String[] args) {
        Coach c = new Coach();
        c.check();
    }
}

interface Performable {
    void play();
}

class Monkey implements Performable {
    @Override
    public void play() {
        System.out.println("Monkey翻跟头");
    }
}

class Lion implements Performable {
    @Override
    public void play() {
        System.out.println("Lion唱摇滚");
    }
}

class Coach {
    public void check() {
        Monkey m = new Monkey();
        m.play();
        Lion l = new Lion();
        l.play();
    }
}

```
