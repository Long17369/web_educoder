import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        Container[] containers = new Container[n];
        for (int i = 0; i < containers.length; i++) {
            String type = scanner.next();
            if (type.equals("cube")) {
                double side = scanner.nextDouble();
                containers[i] = new Cube(side);
            } else if (type.equals("cylinder")) {
                double radius = scanner.nextDouble();
                double height = scanner.nextDouble();
                containers[i] = new Cylinder(radius, height);
            }
        }
        System.out.printf("%.2f\n", Container.sumofArea(containers));
        System.out.printf("%.2f\n", Container.sumofVolume(containers));
        scanner.close();
    }
}

class Cube implements Container {
    private double side;

    public Cube(double side) {
        this.side = side;
    }

    @Override
    public double area() {
        return 6 * side * side;
    }

    @Override
    public double volume() {
        return side * side * side;
    }
}

class Cylinder implements Container {
    private double radius;
    private double height;

    public Cylinder(double radius, double height) {
        this.radius = radius;
        this.height = height;
    }

    @Override
    public double area() {
        return 2 * Container.pi * radius * (radius + height);
    }

    @Override
    public double volume() {
        return Container.pi * radius * radius * height;
    }
}

interface Container {
    public static final double pi = 3.1415926;

    public abstract double area();

    public abstract double volume();

    static double sumofArea(Container c[]) {
        double sum = 0;
        for (Container container : c) {
            sum += container.area();
        }
        return sum;
    }

    static double sumofVolume(Container c[]) {
        double sum = 0;
        for (Container container : c) {
            sum += container.volume();
        }
        return sum;
    }
}
