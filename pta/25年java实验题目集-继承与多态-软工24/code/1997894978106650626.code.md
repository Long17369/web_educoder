import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String plateNo = scanner.next();
        int wheelbase = scanner.nextInt();
        Vehicle vehicle = new Vehicle(plateNo, wheelbase);
        plateNo = scanner.next();
        wheelbase = scanner.nextInt();
        int passengers = scanner.nextInt();
        Bus bus = new Bus(plateNo, wheelbase, passengers);
        vehicle.show();
        bus.show();
        scanner.close();
    }
}

class Vehicle {
    private String plateNo;
    private int wheelbase;

    Vehicle() {
    }

    Vehicle(String plateNo, int wheelbase) {
        this.plateNo = plateNo;
        this.wheelbase = wheelbase;
    }

    public void showInfo() {
        System.out.println("车牌号：" + plateNo);
        System.out.println("轴距：" + wheelbase);
    }

    public void show() {
        System.out.println("车辆信息如下：");
        showInfo();
    }
}

class Bus extends Vehicle {
    private int passengers;

    Bus() {
    }

    Bus(String plateNo, int wheelbase, int passengers) {
        super(plateNo, wheelbase);
        this.passengers = passengers;
    }

    @Override
    public void showInfo() {
        super.showInfo();
        System.out.println("限载人数：" + passengers);
    }

    @Override
    public void show() {
        System.out.println("客车信息如下：");
        showInfo();
    }
}
